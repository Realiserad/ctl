% States are:% idle:% Waiting for a request% % accept_conn:% A new connection has been accepted% % internal_err:% An error occured in the request (e.g. invalid headers received)% Also occurs if the CGI script fails. Puts an error message in the buffer.% % serve_static:% Get the static file content and put it in the buffer% % serve_cgi:% Run the CGI script and put the output in the buffer% % write_data:% Try to write data to the buffer (repeat until empty)% % tcp_fin:% Send the TCP FIN flag to the client when we have no more data% % TLS[	[idle, [accept_conn]],	[accept_conn, [server_static, internal_err, serve_cgi]],	[internal_err, [write_data]],	[serve_static, [write_data]],	[serve_cgi, [internal_err, write_data]],	[write_data, [write_data, tcp_fin]],	[tcp_fin, [idle]]].% Labeling[	[idle, []],	[accept_conn, [p]],	[internal_err, [p,q]],	[serve_static, [p,q]],	[serve_cgi, [p,q]],	[write_data, [p,q,r]],	[tcp_fin, [p]]].% Start stateidle.% Invalid formulaax(ex(and(p,and(q,r))))
